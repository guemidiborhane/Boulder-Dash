3   /**
4    *Represente Parameters which used by CallStatement
5    * the paramters with stocked routines can be IN OUT or INOUT
6    * it is a generic Class
7    * @version 1.0
8    */
9   public class Parameters<T> {
10      /**
11       * Represente Object Parameter
12       * @see Object
13       * */
14      private T obj;
15      /**
16       * Type of this parameter (IN, OUT, INTOUT)
17       * @see TypeParameters
18       * */
19      private TypeParameters typeParameters;
20  
21      /**
22       * Contructor Parameters
23       * @param obj Object of this parameter
24       * @param typeParameters Type of this Parameter OUT, IN, INOUT
25       * @see TypeParameters
26       * @see Object
27       * */
28      public Parameters(final T obj, final TypeParameters typeParameters){
29          this.obj = obj;
30          this.typeParameters = typeParameters;
31      }
32  
33      /**
34       * Set parameter object
35       * @param obj Take any type (Integer, String)
36       * @see Object
37       * */
38      public void setObj(T obj) {
39          this.obj = obj;
40      }
41  
42      /**
43       * Set Type of this Parameter
44       * @param typeParameters IN, OUT, INOUT
45       *  @see TypeParameters
46       * */
47      public void setTypeParameters(TypeParameters typeParameters) {
48          this.typeParameters = typeParameters;
49      }
50  
51      /**
52       * @return Object of this parameter
53       * @see Object
54       * */
55      public T getParameter() {
56          return obj;
57      }
58  
59      /**
60       * @return Type of this Parameter
61       * @see TypeParameters
62       * */
63      public TypeParameters getTypeParameters() {
64          return typeParameters;
65      }
66  }
