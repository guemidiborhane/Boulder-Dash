5   import java.util.ArrayList;
6   import java.util.Collection;
7   
8   /**
9    * The type Reaction define the mechanics when an element hit an other.
10   *
11   * @param <From> the type the element that hit;
12   * @param <To>   the type the element that be hit;
13   */
14  public abstract class Reaction <From extends IAction, To extends IAction> {
15  
16      protected To to;
17      protected From from;
18      private Class<? extends From> fromClass;
19      protected Collection<Sides> sides = new ArrayList<>();
20  
21      /**
22       * Instantiates a new Reaction.
23       *
24       * @param to        the to
25       * @param fromClass the from class
26       * @param sides     the sides that can be hit
27       */
28      Reaction(To to, Class<? extends From> fromClass, Sides[] sides){
29          this.to = to;
30          this.fromClass = fromClass;
31          for (Sides side: sides){
32              this.sides.add(side);
33          }
34      }
35  
36      /**
37       * Is reaction test if and element will react.
38       *
39       * @param from the from
40       * @param side the side
41       * @return the boolean
42       */
43      public boolean isReaction(Object from, Sides side){
44          return (fromClass.isInstance(from) && this.sides.contains(side));
45      }
46  
47      /**
48       * Perform execute the reaction.
49       *
50       * @param from  the from
51       * @param side  the side
52       * @param ticks the ticks
53       * @return the boolean
54       */
55      public boolean perform(Object from, Sides side, int ticks){
56          if (isReaction(from, side)){
57              this.from = (From) from;
58              return this.execute(ticks);
59          }
60          return false;
61      }
62  
63      /**
64       * Execute is the reaction mechanism.
65       *
66       * @param ticks the ticks
67       * @return the boolean
68       */
69      protected abstract boolean execute(int ticks);
70  }
