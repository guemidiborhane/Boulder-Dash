17  /**
18   * The Engine allow to manage the world, entities, sound, display, HUD, etc...
19   * @version 1.0
20   * */
21  public class Engine extends JPanel implements IEngine{
22  
23      /**
24       * The game's world
25       * */
26      private IWorld world;
27  
28      /**
29       * TickGenerator
30       * */
31      private TickGenerator tickGenerator;
32      /**
33       * tread of the tickGenerator
34       * */
35      private Thread tickGeneratorThread;
36      /**
37       * The HUD which display the informations
38       * */
39      private Hud hud;
40  
41      /**
42       * The interface when the game is finish
43       * */
44      private IFinishWorld game;
45      private Timer timer;
46  
47      /**
48       * Engine contructor
49       * @param game Interface when game is finish
50       * */
51      public Engine(IFinishWorld game){
52          this.setLayout(null);
53          this.setFocusable(true);
54          this.requestFocusInWindow();
55  
56          this.tickGenerator = new TickGenerator();
57  
58          this.game = game;
59  
60          // new Sound("ambiance", true).start();
61      }
62  
63      /**
64       * This method allow to load a world, create Hud and it information
65       * and start the tick generator
66       * @param world the world you want load
67       * */
68      public void loadWorld(IWorld world){
69          this.world = world;
70          this.setBackground(Color.black);
71          this.loadComponents();
72          this.configureEntity();
73          this.revalidate();
74          this.repaint();
75  
76          hud = new Hud(new GridLayout(1, 3));
77          this.setComponentZOrder(hud, 0);
78          hud.setSize((int)this.getSize().getWidth(), 30);
79          hud.setBackground(Color.YELLOW);
80  
81          this.timer = new Timer(this, this.hud, this.world.getTimeRemaining());
82          this.tickGenerator.addTickListener(this.timer);
83  
84          hud.addInfo(new Info("Score"));
85          hud.addInfo(new Info("Diamond remaining", this.world.getDiamonds_left()));
86  
87  
88          this.playSound("Start");
89  
90          this.tickGeneratorThread = new Thread(tickGenerator);
91          this.tickGeneratorThread.start();
92      }
93  
94      /**
95       * Allow to unload the world
96       * */
97      private void unloadWorld(){
98          this.world = null;
99          this.hud = null;
100         this.tickGeneratorThread.interrupt();
101         this.tickGenerator.removeAllTickListeners();
102         this.tickGeneratorThread = null;
103         this.removeAll();
104         this.revalidate();
105         this.repaint();
106     }
107 
108     /**
109      * Allow to get the information to the change them
110      *@return Optional(info) if the information it is present return Optional(Info) else Optional.empty()
111      * @see Info
112      * @see Optional
113      * */
114     public Optional<Info> getInfo(final String name){
115         return this.hud.getInfo(name);
116     }
117 
118     @Override
119     public Context getContext(Rectangle rectangle){
120         return new Context(this.world.getComponents(), rectangle);
121     }
122 
123     @Override
124     public void removeEntity(IComponent element) {
125         this.remove((JComponent) element);
126         this.revalidate();
127         this.repaint();
128         this.tickGenerator.removeTickListener((TickListener) element);
129         this.world.removeEntity(element);
130     }
131 
132     @Override
133     public void removeBlock(IComponent element) {
134         this.remove((JComponent) element);
135         this.revalidate();
136         this.repaint();
137         this.world.removeBlock(element);
138     }
139 
140     @Override
141     public void removeCharacter(IComponent element) {
142         this.remove((JComponent) element);
143         this.revalidate();
144         this.repaint();
145         this.world.removeCharacter(element);
146     }
147 
148     @Override
149     public void lose() {
150         this.game.finished(false);
151         this.playSound("Lose");
152         this.unloadWorld();
153     }
154 
155     @Override
156     public void win() {
157         this.game.finished(true);
158         this.playSound("Win");
159         this.unloadWorld();
160     }
161 
162     @Override
163     public boolean isOut(Rectangle projection) {
164         return (
165             0 > projection.x ||
166             0 > projection.y ||
167             this.getSize().width < projection.x + projection.width ||
168             this.getSize().height < projection.y + projection.height
169         );
170     }
171 
172     @Override
173     public boolean levelCompleted() {
174         Optional<Info> d = this.getInfo("Diamond Remaining");
175         if (d.isPresent()){
176             return (d.get().get() <= 0);
177         }
178         return false;
179     }
180 
181     @Override
182     public void playSound(String filename) {
183         new Sound(filename).start();
184     }
185 
186     /**
187      * Allow to loadComponents in engine
188      * the entities recover the engine reference
189      * */
190     private void loadComponents(){
191         Collection<IComponent> components = world.getComponents();
192 
193         for (IComponent component: components){
194             this.add((JComponent) component);
195             component.setEngine(this);
196         }
197     }
198     /**
199      * Allow to configure entities
200      * - add in tickGenerator the entities(TickListerners)
201      * - load the entitie's behaviors
202      * */
203     private void configureEntity(){
204         Collection<IEntity> entities = world.getEntities();
205         //Set Engine for all Entity
206         for(IEntity entity : entities){
207             this.tickGenerator.addTickListener(entity);
208             entity.loadBehaviors();
209         }
210     }
