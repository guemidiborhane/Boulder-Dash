5   import javax.swing.*;

7   import java.util.Collection;
8   import java.util.Optional;

10  /**
11   *Allow to check if the action of a entity is possible, this Thread return the block found
12   * @see Runnable
13   * @see JComponent
14   */
15  public class Context {
16  
17      /**
18       * Represente the virtual rectangle which placed forward the entities which need the context
19       * @see Rectangle
20       * */
21      private Rectangle rectangle;
22      /**
23       * Its a list of all Components in game's map
24       * @see IComponent
25       * */
26      private Collection<IComponent> worldComponents;
27  
28      /**
29       * Allow to get the context when a entity request
30       * @return The context which requested by an entity
31       * @param rectangle Represente a virtual rectangle forward entities
32       * @see Context
33       * */
34      Context(Collection<IComponent> worldComponents, Rectangle rectangle){
35          this.rectangle = rectangle;
36          this.worldComponents = worldComponents;
37      }
38  
39      /**
40       * Allow to get the context this method start the thread context
41       * @return Optional(JComponent) if the context found the block return JComponent else Optional.empty()
42       * @see JComponent
43       * @see Optional
44       * */
45      public Optional<IComponent> get(){
46  
47          for(IComponent component : worldComponents){
48              if(rectangle.intersects(component.getBounds())){
49                  return  Optional.of(component);
50              }
51          }
52          return Optional.empty();
53      }
54  }
