10  /**
11   * Interface for engine

14  public interface IEngine {

16       * Allow to add KeyListerner

20      void addKeyListener(KeyListener keyListener);

22       * Allow to get Context for the entities.
23       * a context is a element forward the entity.

28      Context getContext(Rectangle rectangle);

31       * Allow to remove Entity

33       * */

37       * Allow to remove block
38       * @param element you want add
39       * */
40      void removeBlock(IComponent element);
41

43       * Allow to remove character
44       * @param element Element you want remove
45       * */
46      void removeCharacter(IComponent element);
47  
48      /**
49       * @return  information displayed by HUD
50       * @param name Name of Info Displayed
51       * @see Hud
52       * @see Info

54      Optional<Info> getInfo(final String name);
55  
56      /**
57       * This method is call when player lose the game
58       * (Kill by monster or died)
59       * */
60      void lose();
61  
62      /**
63       * When the player has get all diamond on the map and cross the exit

65       * */
66      void win();

68      /**
69       * @return true if the rectangle is out of false if it is not.
70       * @param projection the rectangle you want test.
71       * */

74      /**
75       * @return true if the level is complete or false if it is not.
76       * */
77      boolean levelCompleted();
78  
79      /**
80       * Allow to play a sound
81       * @param filename Name of the file sound, you want play.
82       * */
83      void playSound(String filename);
